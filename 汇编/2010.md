## 4、输入成绩并排序
```
data segment
    count dw 0
    score db 100 dup(?)
    

data ends
code segment
main proc far
assume cs:code,ds:data,es:data
start:
    push ds
    sub ax,ax
    push ax
    mov ax,data
    mov ds,ax
    mov es,ax

    mov di,0
input:
    mov bx,0
    mov cx,2
inputloop:  
    mov ah,01h
    int 21h
    cmp al,0dh
    je sort
    sub al,30h
    jl exit
    sub ah,ah
    xchg ax,bx
    mov si,10
    mul si
    add bx,ax
    loop inputloop
    mov di,count
    mov byte ptr [score+di],bl
    inc count
    call binihex
    call crlf
    jmp input

    mov bx,count
    call binihex

sort:
    mov cx,count
    dec cx
outloop:
    push cx
    mov di,0
innerloop:
    mov al,[score+di]
    cmp al,[score+di+1]
    jge next
    xchg al,[score+di+1]
    xchg [score+di],al
next:
    inc di
    loop innerloop
    pop cx
    loop outloop


    mov di,0
outputloop:
    mov bl,score[di]
    sub bh,bh
    call binihex
    inc di
    call crlf
    cmp di,count
    jl outputloop

exit:
    ret
main endp
binihex proc near
    mov ch,4
rotate:
    mov cl,4
    rol bx,cl
    mov al,bl
    and al,0fh
    add al,30h
    cmp al,3ah
    jl printit
    add al,07h
printit:
    mov dl,al
    mov ah,02h
    int 21h
    dec ch
    jnz rotate
    ret
binihex endp
crlf proc near
    mov dl,0dh
    mov ah,02h
    int 21h
    mov dl,0ah
    mov ah,02h
    int 21h
    ret
crlf endp
code ends
end start
```
